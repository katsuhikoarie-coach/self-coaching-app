### STEP 2　MySQLテーブル作成
`sql/01_create_table.sql` として以下を作成する。

```sql
-- データベース作成
CREATE DATABASE IF NOT EXISTS yamano_sales
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

USE yamano_sales;

-- 商品テーブル
CREATE TABLE IF NOT EXISTS products (
  id                INT AUTO_INCREMENT PRIMARY KEY,
  product_code      VARCHAR(10)   NOT NULL UNIQUE COMMENT '商品コード',
  product_name      VARCHAR(100)  NOT NULL COMMENT '商品名',
  line_name         VARCHAR(50)   NOT NULL COMMENT 'シリーズ名',
  category          VARCHAR(30)   NOT NULL COMMENT 'スキンケア/メイク/ボディ等',
  product_type      VARCHAR(30)   NOT NULL COMMENT 'クレンジング/化粧水/乳液等',
  price             INT           NOT NULL COMMENT '最新本体価格（円）',
  price_verified_at DATE          NOT NULL COMMENT '価格確認日',
  volume            VARCHAR(20)   NULL     COMMENT '内容量',
  features          TEXT          NULL     COMMENT '商品特徴（3点）',
  talk_points       TEXT          NULL     COMMENT '接客トークポイント・主な成分説明',
  usage_method      TEXT          NULL     COMMENT '使用方法',
  target_concern    VARCHAR(200)  NULL     COMMENT '対象悩み（乾燥/ハリ/毛穴等）',
  target_age        VARCHAR(50)   NULL     COMMENT '対象年齢層',
  is_standard       TINYINT(1)    DEFAULT 0 COMMENT '定番品フラグ（☆マーク）',
  is_featured       TINYINT(1)    DEFAULT 0 COMMENT '注目商品フラグ（積極提案）',
  is_active         TINYINT(1)    DEFAULT 1 COMMENT '販売中フラグ（0=終売）',
  renewal_date      DATE          NULL     COMMENT 'リニューアル予定日',
  notes             VARCHAR(200)  NULL     COMMENT '備考',
  created_at        TIMESTAMP     DEFAULT CURRENT_TIMESTAMP,
  updated_at        TIMESTAMP     DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
```

USE yamano_sales;

-- 件数確認
SELECT COUNT(*) AS 総商品数 FROM products;
SELECT COUNT(*) AS 販売中 FROM products WHERE is_active = 1;
SELECT COUNT(*) AS 注目商品 FROM products WHERE is_featured = 1;

-- ライン別件数
SELECT line_name, COUNT(*) AS 件数 FROM products GROUP BY line_name;

-- 価格一覧確認
SELECT product_code, product_name, price, price_verified_at FROM products ORDER BY line_name, id;
